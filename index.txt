    function renderTables() {
        const selectedFloor = document.getElementById("floorSelect").value;
        const currentFloorQuotas = baseQuotas[selectedFloor];
        const floorData = globalDataStore[selectedFloor];

        const linensBody = document.getElementById("linensBody");
        linensBody.innerHTML = "";
        
        linensItems.forEach((item, index) => {
            const baseValue = currentFloorQuotas[item];
            const state = floorData.linens[item];
            const orderAmount = Math.max(0, baseValue - (state.current - state.expected));

            const row = document.createElement("tr");
            row.innerHTML = `
                <td>${item}</td>
                <td class="base-amount">${baseValue}</td>
                <td><input type="number" id="current-linen-${index}" value="${state.current}" min="0" oninput="updateLinenState(${index}, '${item}')"></td>
                <td><input type="number" id="expected-linen-${index}" value="${state.expected}" min="0" oninput="updateLinenState(${index}, '${item}')"></td>
                <td class="order-amount" id="order-linen-${index}">${orderAmount}</td>
            `;
            linensBody.appendChild(row);
        });

        amenitiesBody.innerHTML = "";

        document.getElementById("quickCmd").value = "";

    function updateLinenState(index, itemName) {
        const selectedFloor = document.getElementById("floorSelect").value;
        const currentVal = parseInt(document.getElementById(`current-linen-${index}`).value) || 0;
        const expectedVal = parseInt(document.getElementById(`expected-linen-${index}`).value) || 0;
        
        globalDataStore[selectedFloor].linens[itemName].current = currentVal;
        globalDataStore[selectedFloor].linens[itemName].expected = expectedVal;

        const baseAmount = baseQuotas[selectedFloor][itemName];
        document.getElementById(`order-linen-${index}`).textContent = Math.max(0, baseAmount - (currentVal - expectedVal));
    }

    function updateAmenityState(index, itemName) {
        const selectedFloor = document.getElementById("floorSelect").value;
        globalDataStore[selectedFloor].amenities[itemName].current = parseInt(document.getElementById(`current-amenity-${index}`).value) || 0;
        globalDataStore[selectedFloor].amenities[itemName].order = parseInt(document.getElementById(`order-amenity-${index}`).value) || 0;
    }

    function processQuickCommand() {
        const inputStr = document.getElementById("quickCmd").value.trim();
        const upperStr = inputStr.toUpperCase();
        const feedback = document.getElementById("quickFeedback");
        
        if (!inputStr) return;

        if (upperStr.startsWith("ALL ")) {
            const payload = inputStr.substring(4).trim();
            const floorBlocks = payload.split('|');
            let successFloors = 0;

            floorBlocks.forEach(block => {
                const parts = block.split(':');
                if(parts.length !== 2) return;
                const floorName = parts[0];
                const dataStr = parts[1];
                if (!globalDataStore[floorName]) return;

                const numbers = dataStr.split('.');
                numbers.forEach((numStr, index) => {
                    if (index >= linensItems.length) return;
                    const parsedNum = parseInt(numStr);
                    if (!isNaN(parsedNum)) {
                        globalDataStore[floorName].linens[linensItems[index]].expected = parsedNum;
                    }
                });
                successFloors++;
            });

            feedback.style.color = "#8e44ad";
            feedback.textContent = `成功載入 ${successFloors} 個樓層的退房加續住數據！`;
            document.getElementById("quickCmd").value = "";
            renderTables();
            setTimeout(() => { feedback.textContent = ""; }, 4000);
            return;
        }

        const parts = inputStr.split(/\s+/);
        if (parts.length < 2) {
            feedback.style.color = "#e74c3c"; feedback.textContent = "格式不完整"; return;
        }

        let targetFloor = document.getElementById("floorSelect").value;
        let cmdIndex = 0;

        const firstPart = parts[0].toUpperCase();
        if (firstPart !== "I" && firstPart !== "A") {
            let floorStr = firstPart.replace('F', '樓');
            if (!floorStr.includes('樓')) floorStr += '樓';

            if (globalDataStore[floorStr]) {
                targetFloor = floorStr;
                cmdIndex = 1;
            } else {
                feedback.style.color = "#e74c3c"; feedback.textContent = "找不到指定的樓層：" + firstPart; return;
            }
        }

        if (parts.length <= cmdIndex + 1) {
            feedback.style.color = "#e74c3c"; feedback.textContent = "格式不完整，缺少指令或數字"; return;
        }

        const cmd = parts[cmdIndex].toUpperCase();
        const numStr = parts[cmdIndex + 1];
        
        if (cmd !== "I" && cmd !== "A") {
            feedback.style.color = "#e74c3c"; feedback.textContent = "請使用 I 或 A 作為指令"; return;
        }

        const numbers = numStr.split('.');
        let count = 0;

        numbers.forEach((num, index) => {
            if (index >= linensItems.length) return;
            const parsedNum = parseInt(num);
            if (isNaN(parsedNum)) return;

            if (cmd === "I") {
                globalDataStore[targetFloor].linens[linensItems[index]].current = parsedNum;
            } else if (cmd === "A") {
                globalDataStore[targetFloor].linens[linensItems[index]].expected = parsedNum;
            }
            count++;
        });

        document.getElementById("floorSelect").value = targetFloor;
        renderTables();

        feedback.style.color = "#27ae60";
        feedback.textContent = `成功寫入 【${targetFloor}】 的 ${count} 筆資料！`;
        document.getElementById("quickCmd").value = "";
        setTimeout(() => { feedback.textContent = ""; }, 3000);
    }

    // ========== 更新：匯出 Excel 時 0 變為空白 ==========
    function exportToExcel(type) {
        const feedback = document.getElementById("exportFeedback");
        const allItems = [...linensItems, ...amenitiesItems];
        let results = [];

        if (type === 'current') {
            const floor = document.getElementById("floorSelect").value;
            allItems.forEach(item => {
                let val = 0;
                if (linensItems.includes(item)) {
                    const base = baseQuotas[floor][item];
                    const state = globalDataStore[floor].linens[item];
                    val = Math.max(0, base - (state.current - state.expected));
                } else {
                    val = globalDataStore[floor].amenities[item].order || 0;
                }
                // 若數值為 0 則替換成空字串，讓 Excel 保持空白
                results.push(val === 0 ? "" : val);
            });
            feedback.innerHTML = `✅ 已成功複製 <strong>【${floor}】</strong> 的 33 項叫貨數量！<br>請前往 Excel 直接貼上即可。`;
            feedback.style.color = "#2980b9";
        } 
        else if (type === 'all') {
            allItems.forEach(item => {
                let totalForThisItem = 0;
                Object.keys(baseQuotas).forEach(floor => {
                    if (linensItems.includes(item)) {
                        const base = baseQuotas[floor][item];
                        const state = globalDataStore[floor].linens[item];
                        totalForThisItem += Math.max(0, base - (state.current - state.expected));
                    } else {
                        totalForThisItem += (globalDataStore[floor].amenities[item].order || 0);
                    }
                });
                // 若總計為 0 則替換成空字串，讓 Excel 保持空白
                results.push(totalForThisItem === 0 ? "" : totalForThisItem);
            });
            feedback.innerHTML = `🚀 已成功加總並複製 <strong>【全館總計】</strong> 的 33 項叫貨數量！<br>請前往 Excel 直接貼上即可。`;
            feedback.style.color = "#d35400";
        }

        const copyStr = results.join('\n');
        
        navigator.clipboard.writeText(copyStr).then(() => {
            feedback.style.display = "block";
            setTimeout(() => { feedback.style.display = "none"; }, 5000);
        }).catch(err => {
            alert("複製失敗，可能是瀏覽器封鎖了剪貼簿權限！\n請手動複製以下內容：\n\n" + copyStr);
        });
    }

        renderTables();
    };
